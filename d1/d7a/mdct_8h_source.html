33  ({ int32_t __hi; \
34  uint32_t __lo; \
35  int32_t __result; \
36  asm ("smull %0, %1, %3, %4\n\t" \
37  "mov %2, %1, lsl #1" \
38  : "=&r" (__lo), "=&r" (__hi), "=r" (__result) \

40  : "cc" ); \
41  __result; \
42  })

46 static inline int32_t fixmul32b(int32_t x, int32_t y)
47 {
48  asm (
49  "mac.l %[x], %[y], %%acc0 \n" /* multiply */
50  "movclr.l %%acc0, %[x] \n" /* get higher half */
51  : [x] "+d" (x)
52  : [y] "d" (y)
53  );

55 }
