83 ini_buffer_t *ini_createBuffer (unsigned int size)
84 {
85  ini_buffer_t *b;
86  /* Allocate memory to structure */
87  if (size == ( ((unsigned) -1 << 1) >> 1 ))
88  return 0; /* Size is too big */
89  b = malloc (sizeof (ini_buffer_t));
90  if (!b)
91  return 0;
92 
93  /* Allocate memory to buffer */
94  b->buffer = malloc (sizeof (char) * (size + 1));
95  if (!b->buffer)
96  {
97  free (b);
98  return 0;
99  }
100  b->size = size;
101 
102  /* Returns address to tcl */
103  return b;
104 }
