30 // The flags below can be used in the `flags` member of the `ddb_cover_query_t` structure,
31 // and can be OR'ed together.
32 //
33 // Example usage: `DDB_ARTWORK_FLAG_NO_FILENAME | DDB_ARTWORK_FLAG_LOAD_BLOB`
34 // This indicates that all results must be loaded into memory, and returned as blob.
35 //
36 // Another example: `DDB_ARTWORK_FLAG_LOAD_BLOB`
37 // This indicates that both blob and filename must be returned.
38 // However, in some cases filenames are not available, e.g. when loading from tags, with disk cache disabled.
39 // In this case filename will be set to NULL.
40 
41 enum {
42  // Tells that filenames should not be returned
43  DDB_ARTWORK_FLAG_NO_FILENAME = 0x00000001,
44 
45  // Returned artwork can be a blob, i.e. a memory block - that is, entire cover image in memory
46  DDB_ARTWORK_FLAG_LOAD_BLOB = 0x00000002,
47 
48  // Don't allow writing files to disk cache, even if the cache is enabled in the settings
49  DDB_ARTWORK_FLAG_NO_CACHE = 0x00000004,
50 };

94  // The `cover_get` function adds the query into an internal queue,
95  // then the queue gets processed on another thread, i.e. asynchronously.
96  //
97  // When the query is processed, the supplied `callback` is called
98  // with the results in `ddb_cover_info_t` structure.
99  //
100  // The callback is guaranteed to be called,
101  // because the caller is responsible for memory management of the `query` argument.
102  //
103  // The callback is not executed on the same thread, as cover_get.
104  // Avoid running slow blocking code in the callbacks.
105  void
106  (*cover_get) (ddb_cover_query_t *query, ddb_cover_callback_t callback);
