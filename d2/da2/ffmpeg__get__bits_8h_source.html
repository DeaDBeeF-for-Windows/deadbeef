138 /* Bitstream reader API docs:
139 name
140  arbitrary name which is used as prefix for the internal variables
141 
142 gb
143  getbitcontext
144 
145 OPEN_READER(name, gb)
146  loads gb into local variables
147 
148 CLOSE_READER(name, gb)
149  stores local vars in gb
150 
151 UPDATE_CACHE(name, gb)
152  refills the internal cache from the bitstream
153  after this call at least MIN_CACHE_BITS will be available,
154 
155 GET_CACHE(name, gb)
156  will output the contents of the internal cache, next bit is MSB of 32 or 64 bit (FIXME 64bit)
157 
158 SHOW_UBITS(name, gb, num)
159  will return the next num bits
160 
161 SHOW_SBITS(name, gb, num)
162  will return the next num bits and do sign extension
163 
164 SKIP_BITS(name, gb, num)
165  will skip over the next num bits
166  note, this is equivalent to SKIP_CACHE; SKIP_COUNTER
167 
168 SKIP_CACHE(name, gb, num)
169  will remove the next num bits from the cache (note SKIP_COUNTER MUST be called before UPDATE_CACHE / CLOSE_READER)
170 
171 SKIP_COUNTER(name, gb, num)
172  will increment the internal bit counter (see SKIP_CACHE & SKIP_BITS)
173 
174 LAST_SKIP_CACHE(name, gb, num)
175  will remove the next num bits from the cache if it is needed for UPDATE_CACHE otherwise it will do nothing
176 
177 LAST_SKIP_BITS(name, gb, num)
178  is equivalent to LAST_SKIP_CACHE; SKIP_COUNTER
179 
180 for examples see get_bits, show_bits, skip_bits, get_vlc
181 */

595 {\
596  int n, nb_bits;\
597  unsigned int index;\
598 \
599  index= SHOW_UBITS(name, gb, bits);\
