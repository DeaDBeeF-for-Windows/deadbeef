297 inline void MOS6510::clock (void)
298 {
299  int_least8_t i = cycleCount++;
300 
301 #if 0
302  // C++ exception version
303  try {
304  (this->*procCycle[i]) ();
305  } catch (int_least8_t delta) {
306  cycleCount += delta;
307  m_blocked = true;
308  eventContext.cancel (this);
309  }
310 #endif
311 
312 #if 0
313  // longjmp version
314  int_least8_t delta = setjmp (jmp_env);
315  if (delta == 0) {
316  (this->*procCycle[i]) ();
317  }
318  else {
319  cycleCount += delta;
320  m_blocked = true;
321  eventContext.cancel (this);
322  }
323 #endif
324 
325 #if 1
326  if (!rdy || !aec) {
327  m_stealCycleDelta = -1;
328  }
329  else {
330  (this->*procCycle[i]) ();
331  }
332  if (m_stealCycleDelta != 0) {
333  cycleCount += m_stealCycleDelta;
334  m_stealCycleDelta = 0;
335  m_blocked = true;
336  eventContext.cancel (this);
337  }
338 #endif
339 }
