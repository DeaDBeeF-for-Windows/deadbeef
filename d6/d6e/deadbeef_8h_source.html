44 // every plugin must define the following entry-point:
45 // extern "C" DB_plugin_t* $MODULENAME_load (DB_functions_t *api);
46 // where $MODULENAME is a name of module
47 // e.g. if your plugin is called "myplugin.so", $MODULENAME is "myplugin"
48 // this function should return pointer to DB_plugin_t structure
49 // that is enough for both static and dynamic modules
50 
51 // backwards compatibility is supported since API version 1.0
52 // that means that the plugins which use the API 1.0 will work without recompiling until API 2.0.
53 //
54 // increments in the major version number mean that there are API breaks, and
55 // plugins must be recompiled to be compatible.
56 //
57 // add DDB_REQUIRE_API_VERSION(x,y) macro when you define the plugin structure
58 // like this:
59 // static DB_decoder_t plugin = {
60 // DDB_REQUIRE_API_VERSION(1,0)
61 // ............
62 // }
63 // this is required for versioning
64 // if you don't do it -- no version checking will be done (useful for debugging/development)
65 //
66 // please DON'T release plugins without version requirement
67 //
68 // to ensure compatibility, use the following before including deadbeef.h:
69 // #define DDB_API_LEVEL x
70 // where x is the minor API version number.
71 // that way, you'll get errors or warnings when using incompatible stuff.
72 //
73 // if you also want to get the deprecation warnings, use the following:
74 // #define DDB_WARN_DEPRECATED 1
75 //
76 // NOTE: deprecation doesn't mean the API is going to be removed, it just means
77 // that there's a better replacement in the newer deadbeef versions.
