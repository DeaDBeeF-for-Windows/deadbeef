58 /*
59  * Waits for a file descriptor/socket to change status.
60  *
61  * users can use this handler to provide their own implementations,
62  * for example abortable ones
63  *
64  * params :
65  * data whatever parameter may be needed by implementation
66  * fd file/socket descriptor
67  * state MMS_IO_READ_READY, MMS_IO_WRITE_READY
68  * timeout_sec timeout in seconds
69  *
70  *
71  * return value :
72  * MMS_IO_READY the file descriptor is ready for cmd
73  * MMS_IO_ERROR an i/o error occured
74  * MMS_IO_ABORTED command aborted
75  * MMS_IO_TIMEOUT the file descriptor is not ready after timeout_msec milliseconds
76  * every other return value is interpreted same as MMS_IO_ABORTED
77  */
78 typedef int (*mms_io_select_func)(void *data, int fd, int state, int timeout_msec);
