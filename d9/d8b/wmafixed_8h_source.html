78 {

80 #warning Coldfire fixmul32() only works for PRECISION == 16
81 #endif
82  int32_t t1;
83  asm (
84  "mac.l %[x], %[y], %%acc0 \n" // multiply
85  "mulu.l %[y], %[x] \n" // get lower half, avoid emac stall
86  "movclr.l %%acc0, %[t1] \n" // get higher half
87  "lsr.l #1, %[t1] \n"
88  "move.w %[t1], %[x] \n"
89  "swap %[x] \n"
90  : [t1] "=&d" (t1), [x] "+d" (x)
91  : [y] "d" (y)
92  );
