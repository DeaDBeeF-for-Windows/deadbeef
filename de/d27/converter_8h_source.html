227  // The 'get_output_path' function should be used to get the fully
228  // qualified output file path, to be passed to 'convert' function.
229  // It is commonly used by converter GUI.
230  // Parameters:
231  // it: the track
232  // outfolder: the folder to write the file to (usually specified in GUI)
233  // outfile: the filename pattern, which may include additional folder
234  // structure, and title formatting; without extension.
235  // examples: "%artist% - %title%", "subfolder/%title%"
236  // encoder_preset: an existing encoder preset.
237  // preserve_folder_structure: set to 1 to recreate the existing folder
238  // structure, when converting multiple files
239  // root_folder: common root path of all the tracks being converted in 1 go.
240  // write_to_source_folder: set to 1 to write output to the same folders
241  // where input files are located.
242  // out: the buffer for the output file path,
243  // which will come out not escaped, will include the file extension.
244  // sz: size of the out buffer.
245  void
246  (*get_output_path) (DB_playItem_t *it, const char *outfolder, const char *outfile, ddb_encoder_preset_t *encoder_preset, int preserve_folder_structure, const char *root_folder, int write_to_source_folder, char *out, int sz);
