42 // every plugin must define the following entry-point:
43 // extern "C" DB_plugin_t* $MODULENAME_load (DB_functions_t *api);
44 // where $MODULENAME is a name of module
45 // e.g. if your plugin is called "myplugin.so", $MODULENAME is "myplugin"
46 // this function should return pointer to DB_plugin_t structure
47 // that is enough for both static and dynamic modules
48 
49 // backwards compatibility is supported since API version 1.0
50 // that means that the plugins which use the API 1.0 will work without recompiling until API 2.0.
51 //
52 // increments in the major version number mean that there are API breaks, and
53 // plugins must be recompiled to be compatible.
54 //
55 // add DDB_REQUIRE_API_VERSION(x,y) macro when you define the plugin structure
56 // like this:
57 // static DB_decoder_t plugin = {
58 // DDB_REQUIRE_API_VERSION(1,0)
59 // ............
60 // }
61 // this is required for versioning
62 // if you don't do it -- no version checking will be done (useful for debugging/development)
63 //
64 // please DON'T release plugins without version requirement
65 //
66 // to ensure compatibility, use the following before including deadbeef.h:
67 // #define DDB_API_LEVEL x
68 // where x is the minor API version number.
69 // that way, you'll get errors or warnings when using incompatible stuff.
70 //
71 // if you also want to get the deprecation warnings, use the following:
72 // #define DDB_WARN_DEPRECATED 1
73 //
74 // NOTE: deprecation doesn't mean the API is going to be removed, it just means
75 // that there's a better replacement in the newer deadbeef versions.
